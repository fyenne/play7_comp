def lag_feature( df, lags, cols):
    for col in cols:
        print(col)
        tmp = df[['destsitecode',col ]]
        for i in lags:
            shifted = tmp.copy()
            shifted.columns = ['destsitecode', col + "_lag_"+str(i)]
            shifted.date_block_num = shifted.date_block_num + i
            df = pd.merge(df, shifted, on=['destsitecode'], how='left')
    return df
